نُسخۂ طُورِ سِیناء
انجیلِ جلیل کے مختلف حِصص پہلے بُردی1 سے بَنے الگ الگ طُوماروں پر تحریر کئے جاتے اور رکھے جاتے تھے۔ بعد ازاں، جب رقّ2 پر کتابیں مرتّب کرنے کا فنّ وجود میں آیا، تو اُنھیں ایک ہی جِلد میں جمع کرنے کا رواج بھی وجود میں آیا۔ لہٰذا نُسخۂ طُورِ سِیناء میں انجیل کے مختلف حِصص یعنی صحائف/ سورتوں کو ایک جِلد میں ایک جگہ جمع کر دیا گیا، مگر اِن حِصص کے اپنے اپنے جداگانہ نام برقرار رکھے گئے۔ نُسخۂ طُورِ سِیناء میں بھی یہ روایتی نام حاشئے میں لکھے مِلتے ہیں۔ اِن ناموں کی بناء پر تابعین محتلف حِصوں کا حوالہ دے سکتے تھے۔یہ تو سب جانتے ہیں کہ ترجمہ کسی نہ کسی متن کا ہوتا ہے۔ اگر متن ہی ناقابلِ اعتبار ہو، تو ترجمہ چاہے کتنا ہی خوب صورت کیوں نہ کیا جائے، معتبر نہیں ٹھہر سکتا۔ کسی فارسی شاعر نے کہا ہے؎خشتِ اوّل چون نہد معمار کجتا ثریا می رود دیوار کج(اگر معمار پہلی ہی اینٹ ٹیڑھی رکھ دے، تو دیوار ثریا [ستارے] تک ٹیڑھی ہی چلی جائےگی۔)اِس لئے مستند اور معتبر ترجمہ کرنے کے لئے لازم تھا کہ ہم کسی ایسے قلمی نسخے کا انتخاب کرتے جو مستند اور معتبر ہو۔ پس ہم نے کسی ترجمے سے ترجمہ کرنے کی بجائے انجیلِ جلیل کا وہ قدیم یونانی3 نسخہ استعمال کیا ہے جو ”نسخۂ سِیناء“ (Sinaiticus) کے نام سے جانا جاتا ہے اور برطانیہ کے کتاب گھر (برٹش لائبریری British Library) میں موجود ہے۔ یہ حلال جانوروں کی کھالوں پر تحریر ہے۔ یہ نسخہ جو زمانۂ آمدِ اسلام سے پیشتر کا ہے، اُن نسخوں میں سے ایک ہے جو قدیم‌ترین گردانے جاتے ہیں۔ چوتھی صدی عیسوی کا یہ نسخہ — یعنی آنحضرت صلّی اللّٰه علیہ وسلّم کی ولادتِ باسعادت سے بھی تین سَو برس پہلے کا قلمی نسخہ — اُن حلقوں میں جو قدیم نسخوں کی جانچ پڑتال کرتے ہیں، عبرانی زبان کے پہلے حرف א (الف) یا انگریزی زبان کے پہلے نمبر 01 کے نشان سے جانا جاتا ہے۔ ہم نے کوئی چھَپا ہؤا نسخہ استعمال کرنے کی بجائے کمپیوٹر سے تیارکردہ اصل نسخے کے بعینہٖ رنگین تصویری عکس سے استفادہ کیا ہے۔چوں کہ آپ کے ہاتھوں میں جو تفسیرِ انجیل جلیل ہے یہ نسخۂ طُورِ سِیناء پر مبنی ہے، اِس لئے ہم نے مناسب سمجھا ہے کہ اُس کا مختصر سا تعارف کروا دیں۔ہم میں سے بھلا کون ایسا ہوگا جو ”کوہِ طُور“ کے نام سے واقف نہ ہو؟ ہمارے ہاں تو حضرتِ مُوسٰیؑ اور ”کوہِ طُور“ کا چولی دامن کا ساتھ ہے۔ رہی سہی کسر شُعرائے کرام نے پوری کر دی۔ بقول غالب؎کیا فرض ہے کہ سب کو ملے ایک سا جوابآؤ نا ہم بھی سیر کریں کوہِ طُور کییہ الگ بات ہے کہ ہم اِس حقیقت سے بےخبر دکھائی دیتے ہیں کہ ”پہاڑ“ کو فارسی میں”کوہ“، عربی میں”جَبَلٌ“، عبرانی میں”صُور“ [סוּר] اور آرامی میں ”صُوراء“ [סוּרָא] کہتے ہیں جس کی عربی شکل”طُورٌ“ ہے۔ سُریانی میں یہ لفظ ہے ہی”طُور“۔ قرآنِ مجید میں لفظ ”طُور“ حسبِ ذیل دس (10) مقامات پر آیا ہے:سورۃُ البقرۃ [2] کی آیات 63 اور 93؛ سورۃُ النساء [4] کی آیت 154؛ سورۃ مریم [19] کی آیت 52؛ سورۃ طٰہٰ [20] کی آیت 80؛ سورۃ المؤمنون [23] کی آیت 20؛ سورۃُ القصص [28] کی آیات 29 اور 46؛ سورۃُ الطُّور [52] کی آیت 1؛ سورۃُ التِّین [95] کی آیت 2۔غور طلب بات یہ ہے کہ قرآنِ مجید میں دس (10) میں سے آٹھ (8) مقامات پر یہ لفظ حرفِ تخصیص ”اَلْ“ کے ساتھ آیا ہے جو اسمائے عَلَم کے ساتھ نہیں لگایا جاتا۔ اِسی سے معلوم ہو جاتا ہے کہ یہ اسم معرفہ نہیں، بلکہ اسم نکرہ ہے۔ قرآنِ مجید میں دو (2) مقامات پر (سورۃُ المؤمنون [23]، آیت 20 اور سورۃُ التِّین [95]، آیت 2 میں) یہ اسم عَلَم کے طَور پر آیا ہے۔ سورۃُ المؤمنون میں ”طُورِ سِینَاء“ کی شکل میں اور سورۃُ التِّین میں ”طُورِ سِینِینَ“ کی صُورت میں اسمِ عَلَم کے ایک جُزو کی حیثیت سے آیا ہے۔ یہاں”سِینَاء“ کو”سِینِینَ“ بنا دینے کی وجہ یہ دکھائی دیتی ہے کہ اِس سے پہلے ”وَالتِّینِ“، ”وَالزَّیْتُونِ“ اور بعد ازاں ”الْبَلَدِ الْأَمِینِ“ لائے گئے ہیں، اِسی لئے قوافی کی مناسبت کا تقاضا یہ تھا کہ یہ لفظ اِس صُورت میں تحریر کیا جاتا۔آج کَل یہ مقام عربی مصادر میں — حضرتِ مُوسٰیؑ کی مناسبت سے — ”جَبَلُ مُوسٰیؑ“ کے نام سے جانا جاتا ہے۔ یہ موجودہ مصر کے جزیرہ نمائے سِیناء میں واقع ہے۔ اہلِ کتاب (یہودی اور مسیحی) اور اہلِ اسلام کی روایات کے مطابق یہ وہی پہاڑ ہے جہاں حضرتِ مُوسٰیؑ کو تجلّئ الٰہی نصیب ہوئی اور جہاں پروردگارِ عالَم نے آپؑ کو توراۃِ صد صفات عطا فرمائی یا کم از کم احکامِ عشرہ دئے۔ہم نے اوپر جزیرہ نمائے سِیناء کے جس پہاڑ کا ذکر کیا ہے، وہیں وہ خانقاہ واقع ہے جو عربی میں”دَیرِ سِیناء“ اور یُونانی میں μονὴ τῆς ἁγίας αἰκατερίνης [”مونے تیس ھَگیاس اَیْکَاتَیْرِینیس“] کے نام سے جانی جاتی ہے۔ یونیسکو کے مطابق یہ دنیا کی قدیم‌ترین خانقاہ ہے۔اِس خانقاہ کا کتب‌خانہ (لائبریری) عدیم المثال ہے۔ یہاں یونانی، عربی، آرمینی، عبرانی، جارجیانی، سُریانی اور اُدِی(Udi) زبانوں کے بےشمار قلمی نسخے موجود ہیں۔
حضور صلّی اللّٰه علیہ وسلم کا عہدنامہ
ہمارے نکتۂ نظر سے سب سے اہم تحریر وہ ”عہدنامہ“ یا معاہدہ ہے جس کے بارے میں کہا جاتا ہے کہ اِسے جنابِ رسولُ اللّٰه — صلّی اللّٰه علیہ وسلّم — نے اس خانقاہ میں رہنے والے رہبان کو ؁628ء میں عطا فرمایا تھا۔ اِس کی ایک نقل استنبول (ترکی) کے مشہور عجائب گھر (museum) ”توپ کاپی“ (Topkapi) میں محفوظ ہے۔ جب راقمُ الحروف اِس کی زیارت کی سعادت حاصل کرنے ؁2010ء میں وہاں پہنچا، تو تبرکات کی انچارج آنسہ طُوبٰی قُرتلُش (Tuba Qurtuluş) نے بتایا کہ مخدوش حالت میں ہونے کی بناء پر اِسے وہاں سے ہٹا دیا گیا ہے اور محفوظ کرنے کے لئے دوسری جگہ منتقل کر دیا گیا ہے۔ راقمُ الحروف کی اِس سلسلے میں آنسہ طُوبٰی قُرتلُش سے ابھی تک خط و کتابت جاری ہے۔
عکسی تصویر
اِس ”عہدنامہ“ کی ایک نقل دَیرِ سِیناء میں بھی محفوظ ہے۔ ہم یہاں اُس کی عکسی تصویر (فوٹو کاپی photocopy) پیش کرنے کی سعادت حاصل کر رہے ہیں۔ہمیں از حد افسوس ہے کہ ہم اِن سے زیادہ واضح تصاویر حاصل نہیں کر پائے۔ توپ کاپی عجائب‌گھر کی انگریزی اور ترکی زبانوں میں چھَپی ہوئی کتابوں میں بھی اِس کی کوئی واضح‌تر تصویر نہیں مل پائی۔
ترجمہ
ڈاکٹر اکرم ظہور صاحب اور ڈاکٹر ضیاءُ الحق صاحب نے اپنی کتاب ”مسلم ہسٹری 570‏ء سے ؁1950ء تک“4 میں اِس عبارت کا جو ترجمہ انگریزی میں پیش کیا ہے، اُس کا اردو میں خلاصہ ”ساگر ٹائمز“ میں شائع ہؤا تھا۔ ہم اُسے یہاں نقل کرنے کی سعادت حاصل کر رہے ہیں:یہ محمدؐ ابن عبدُ اللّٰه کا پیغام ہے، بطور معاہدہ، دور و نزدیک کے عیسائیوں کے نام۔ بیشک مَیں، میرے ساتھی اور میرے پیروکار اِن کا دفاع کرتے ہیں کیوں کہ عیسائی میری رعایا ہیں اور خدا کی قسم مَیں انہیں ناخوش کرنے والی ہر بات کے خلاف ہوں۔ اُن پر کوئی جبر نہیں۔ اُن کے منصفین کو اُن کے عہدوں سے ہٹایا نہیں جائےگا اور نہ ہی اُن کے راہبوں کو اُن کی خانقاہوں سے نکالا جائےگا۔ کوئی بھی ان کی عبادت‌گاہوں کو تباہ نہیں کرےگا، نقصان نہیں پہنچائےگا، اور نہ ہی یہاں سے کوئی چیز اٹھا لے جائےگا۔ اِن میں سے کسی بھی اقدام کا مرتکب ہونے والا خدا کے معاہدے کو توڑےگا اور اُس کے پیغمبر کی نافرمانی کرےگا۔ بےشک یہ میرے اتحادی ہیں اور میری نصرت رکھتے ہیں ان باتوں کے خلاف جو انہیں ناخوش کریں۔ اُنہیں سفر یا جنگ پر مجبور نہیں کیا جائےگا۔ اُن کی جگہ مسلمان لڑیں‌گے۔ اگر کوئی مسلمان کسی عیسائی خاتون سے عقد کرنا چاہے، تو یہ خاتون کی اجازت کے بغیر ناہوگا۔ اُسے اپنی عبادت‌گاہ میں جا کر عبادت کرنے سے روکا نہیں جائےگا۔ اُن کی عبادت‌گاہوں کا احترام کیا جائےگا۔ میرا کوئی بھی امتی روزِ قیامت تک اِس معاہدے کی خلاف ورزی نہ کرے۔5
نسخۂ سِیناء کی تاریخ
اِس ضمن میں ایک بات بالکل طےشدہ ہے کہ نسخۂ سِیناء ؁360ء کے بعد کا نہیں ہو سکتا کیوں کہ اِس کے حواشی میں جن بزرگوں کے حوالے دئے گئے ہیں، وہ سب کے سب ؁360ء سے پہلے وفات پا چکے تھے۔ اغلب ؁325ء کی تصدیق اِس بات سے بھی ہوتی ہے کہ قسطنطینِ اعظم نے دینِ عیسوی قبول کرنے کے بعد ؁331ء میں اِس بات کا حکم دیا کہ کلامُ اللّٰه کی پچاس نقول سرکاری خرچے پر تیار کی جائیں اور سلطنت کے مختلف مراکز کو بھیج دی جائیں۔؁1761ء میں ایک اطالوی سیاح وِتالیانو دوناتی (Vitaliano Donati) اپنے سفرنامے میں اِس بات کا ذکر کرتا ہے کہ وہ دَیرِ سِیناء پہنچا اور وہاں ایک خوب‌صورت نسخہ دیکھا۔ یوں لگتا ہے کہ وہ نسخۂ سِیناء تھا۔ افسوس کی بات ہے کہ باہر کی دنیا اِس نسخے کے وجود سے ناواقف رہی یہاں تک کہ ؁1844ء میں ایک جرمن عالم تِشَنْدورْف(Tischendorf) نے اِس دَیر اور اُس کی لائبریری کی زیارت کی۔ آخرِ کار اُس نے دَیرِ سِیناء کے رہبان کو اِس بات پر آمادہ کر لیا کہ یہ نسخہ زارِ روس اِسکندرِ دوئم کی خدمت میں تحفتًا پیش کر دیا جائے۔ انقلابِ روس کے برسوں بعد یعنی ؁1933ء میں برٹش میوزیم نے یہ نسخہ سوویت روس کی حکومت سے خرید لیا۔آج کَل یہ نسخہ (جو حلال جانوروں کی کھالوں پر تحریر ہے اور جسے عربی میں ”رَقّ“6 کہتے ہیں برٹش لائبریری (British Library) میں خاص و عام کو دعوتِ نظارہ و مطالعہ دے رہا ہے۔ اُنھوں نے اِس نسخے کی نقل www.codexsinaiticus.org/ ویب سائٹ پر بھی ڈال دی ہے تاکہ علمائے کرام اِس سے مستفید ہو سکیں۔ یونانی متن ہم نے کوئی چھَپا ہؤا نسخہ استعمال کرنے کی بجائے کمپیوٹر سے تیارکردہ اصل نسخے کے بعینہٖ رنگین تصویری عکس سے استفادہ کیا ہے۔
اِس نسخے کے بارے میں کچھ چیدہ چیدہ باتیں
ابھی تک انجیلِ جلیل کا اِتنا قدیم اور مکمل نسخہ دریافت نہیں ہؤا۔ یہ نہ صرف قدیم‌ترین اور مکمل‌ترین نسخہ ہے، بلکہ پروردگار کا کرنا دیکھئے کہ جہاں اللّٰه تعالٰی نے حضرتِ موسٰیؑ کو توراۃِ صد صفات عطا کی
