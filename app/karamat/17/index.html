کراماتِ حواریینؓ رکوع 17
دشمنانِ دِین کی طرف سے سِلونِیکہ شہر میں ہنگامہ‌آرائی
حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ اگلے شہر1 اَمپفولیہ2 اور اَپلونِیہ3 سے ہوتے ہوئے سِلونِیکہ4 شہر پہنچے۔ وہاں پر بنی اسرائیل کی ایک مسجد تھی۔1آپؓ حسبِ معمول مسجد تشریف لے گئے۔ تین سبت اُنھیں کتابُ اللّٰه میں سے دلائل دیتے2اور پکی دلیلوں سے ثابت کرتے رہے کہ ”خلیفۃُ اللّٰه علیہ السلام کا دکھ اٹھانا اور شہادت کے بعد جی اٹھنا لازمی امر تھا۔ لہٰذا مَیں جس عیسٰیؑ کی حقانیت کی شہادت دے رہا ہوں، وہی موعود خلیفۃُ اللّٰه ہیں۔“3چناں چہ بعض اُن بنی اسرائیلی سامعین میں سے قائل ہو کر مولا عیسٰیؑ پر ایمان لائے اور حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ کے ہمنوا ہو گئے۔ اُن کے علاوہ اللّٰه تعالٰی کو ماننے والے غیرِ بنی اسرائیل کی ایک بڑی تعداد — جس میں بہت سی باوقار خواتین بھی شامل تھیں — مولا عیسٰیؑ پر ایمان لے آئی۔4دعوتِ دِین کی یہ صورتِ حال دیکھ کر باقی یہودی حسد میں مبتلا ہو گئے۔5 اُنھوں نے چند بازاری بدمعاش ہمراہ لئے اور شہر میں ہنگامہ بپا کر دیا۔ پھر حضرت یاسینؒ6 کے گھر پر ہلّہ بول دیا تاکہ حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ کو تلاش کر کے ہنگامہ‌خیزوں کے ازدحام میں لے آئیں۔5جب وہ دونوں اُنھیں وہاں نہ ملے، تو وہ حضرت یاسینؒ اور بعض مومن بھائیوں کو شہری حکام کے سامنے لے گئے اور چیخ چیخ کر پکارنے لگے کہ ”یہ لوگ دنیا بھر میں فساد پھیلاتے ہوئے یہاں تک بھی آ گئے ہیں۔6اِس یاسینؒ نے اُن کی مہمان‌نوازی کی۔ یہ سب بادشاہِ وقت7 کے باغی ہیں۔ یہ تو کسی اَور کو بادشاہ کہتے ہیں جس کا نام ’عیسٰیؑ‘ بتاتے ہیں۔ اِس یاسینؒ نے اُنھیں اپنے گھر میں پناہ دی ہوئی ہے۔“7اِس صورتِ حال کے پیشِ نظر سب لوگ پریشان ہو گئے اور یہ باتیں سن کر شہری حکام بھی۔8پھر شہری حکام نے حضرت یاسینؒ اور اُن کے ساتھیوں کو مُچلکے پر رِہا کر دیا۔9
حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ کا وارِیہ شہر میں درس
مومن بھائیوں نے رات ہوتے ہی حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ کو شہرِ وارِیہ8 بھجوا دیا۔ وہاں پہنچ کر یہ حضرات وہاں کی مسجدِ بنی اسرائیل تشریف لے گئے۔10یہاں کے بنی اسرائیلی لوگ سِلونِیکہ شہر کے بنی اسرائیلی لوگوں کی نسبت کافی حد تک حق کے قبول کرنے والے تھے۔ وہ بڑے دھیان سے حضرت طالوتِ اصغرؓ اور حضرت سِلوانؒ کی تعلیمات سنتے تھے۔ وہ ہر روز کلامُ اللّٰه کی روشنی میں تحقیق کرتے کہ ”کیا یہ امرِ واقعی ہے جس کی طرف ہماری رہنمائی کی جا رہی ہے؟“11اِس کے نتیجے میں بہت سارے بنی اسرائیلی ایمان لے آئے اور غیرِ بنی اسرائیل میں سے بھی بہت ساری معزز خواتین اور مرد۔12جب سِلونِیکہ شہر کے یہودی علمائے سُوء کو معلوم ہؤا کہ ”طالوتِ اصغرؓ شہرِ وارِیہ میں دعوت دیتا پھر رہا ہے“، تو اُنھوں نے وہاں پہنچ کر لوگوں کو آپؓ کی مخالفت پر اکسایا۔13پھر مومن بھائیوں نے حضرت طالوتِ اصغرؓ کو فورًا ہی سمندر کی طرف رخصت کر دیا، لیکن حضرت سِلوانؒ اور حضرت تمجیدُ اللّٰهؒ وہیں وارِیہ میں رہ گئے۔14حضرت طالوتِ اصغرؓ کی رہنمائی کرنے والے آپؓ کو بحری جہاز کے ذریعے شہرِ ایتھِنز لے گئے۔ پھر حضرت سِلوانؒ اور حضرت تمجیدُ اللّٰهؒ کے لئے حضرت طالوتِ اصغرؓ کا یہ پیغام لے کر واپس آ گئے کہ ”میرے پاس جلد از جلد آ جاؤ!“15
شہرِ ایتھِنز میں شرک کے بارے میں مباحثے
جب حضرت طالوتِ اصغرؓ اُن کے واپس آنے کا شہرِ ایتھِنز میں انتظار کر رہے تھے، تو اہالئ شہر کو بت‌پرستی میں مبتلا دیکھ کر آپؓ بہت رنجیدہ خاطر ہوئے۔16لہٰذا اِس موضوع پر آپؓ مسجد میں بنی اسرائیلی یا خوفِ خدا رکھنے والے غیرِ بنی اسرائیلی لوگوں اور بازار میں جن اَور لوگوں سے ملتے، اُن سے بحث مباحثہ کیا کرتے۔17اِسی طرح کئی فلاسفۂ اَبقُورِیِّین9 و رواقِیِّین10 بھی آپؓ سے بحث کیا کرتے۔ بعض تو — نعوذ باللّٰه! — یہاں تک کہتے کہ ”اِس نے اِدھر اُدھر سے کچھ علم چَر چُگ لیا ہے“ اور بعض یہ کہتے کہ ”یہ تو ایسے دیوتاؤں کا ذکر کرتا ہے جن کا نام ہم نے آج تک نہیں سنا۔“ وہ یہ اِس لئے کہتے تھے کیوں کہ آپؓ سیدنا عیسٰی علیہ السلام اور یومِ قیامت کا ذکر کرتے رہتے تھے۔1118وہ لوگ آپؓ کو ایتھِنز شہر کے عوامی جرگے لے گئے۔ وہاں آپؓ سے یہ کہا گیا کہ ”ہمیں ذرا اپنے نئے عقیدے کے متعلق تو بتاؤ!19یہ جو تم نئی نئی باتیں کہتے پھر رہے ہو، اِن کا مقصد کیا ہے؟“20اُنھوں نے یہ اِس لئے کہا کیوں کہ اہلِ ایتھِنز اور مقیمی پردیسی نئے نئے افکار کے بارے میں بحث مباحثہ کرنے میں اپنا وقت صَرف کیا کرتے تھے۔21
شہرِ ایتھِنز کے عوامی جرگے میں حضرت طالوتِ اصغرؓ کا اپنے دفاع میں بیان
حضرت طالوتِ اصغرؓ عوامی جرگے میں کھڑے ہو کر فرمانے لگے: ”اہلِ ایتھِنز! آپ حضرات مجھے ہر طرح سے بڑے مذہبی دکھائی دیتے ہیں22کیوں کہ مَیں نے اِس شہر میں آتے ہوئے آپ لوگوں کے تیرتھوں میں ایک ایسا استھان بھی دیکھا جس پر ’نامعلوم دیوتا‘ کے الفاظ کندہ تھے۔ پس آپ لوگ جس خدا کی حقیقت جانے بغیر پوجا کئے چلے جا رہے ہیں، مَیں اُسی کی معرفت سے آپ لوگوں کو آگاہ کرتا ہوں۔23خدا نے دنیا و ما فیہا کو خلق کیا ہے۔ وہی زمین و آسمان کا مالک ہے۔ وہ انسانوں کے بنائے ہوئے مندروں کا محتاج نہیں۔24وہ خدا نہ تو انسانی خدمت کا محتاج ہے، نہ ہی کسی چیز کا ضرورت‌مند۔ وہ تو سب کو زندگی اور جان بخشتا اور سب کچھ عطا کرتا ہے۔25اُسی نے تمام اقوامِ عالَم کو ایک ہی وجود سے پیدا کیا تاکہ انسان تمام رُوئے زمین کو آباد کریں۔ اُس نے اُن کے لئے اوقات و مقامات کا تعین کر دیا کہ کون سی قوم کس جگہ پر رہے اور کتنے عرصے کے لئے رہے۔1226مقصد یہ تھا کہ وہ اُس خدا کی تلاش میں رہیں۔ اِس طرح ممکن ہے کہ وہ اُس تک پہنچ جائیں — اگرچہ وہ ذاتِ لاشریک شہ رگ سے بھی قریب ہے۔27تمھارے ایک شاعر نے بھی کہا ہے کہ ’اُس کی ذاتِ اقدس کی بدولت ہم زندہ اور سرگرم ہیں۔‘13 تمھارے ایک دوسرے شاعر کا بیان ہے کہ ’ہمارے وجود کا منبع و مخرج اُسی کی ذاتِ اقدس ہے۔‘1428”اب جب کہ ہمارا وجود خدا کی ذاتِ اقدس پر مبنی ٹھہرا، تو ہمیں چاہئے کہ خدائے متعال کی ذاتِ پاک کو پتھر، چاندی یا سونے کے بتوں پر قیاس نہ کریں جو انسانی ذہن اور ہاتھوں کی تخلیق ہیں۔29زمانۂ سابق میں خدا تعالٰی اِس جہالت کو نظر انداز کرتا رہا، لیکن اب ہر جگہ کے رہنے والے سب لوگوں کو توبہ کا حکم دیتا ہے۔30اُس نے یومِ حساب مقرر کیا ہے جس دن وہ تمام بنی نوعِ انسان کا فیصلہ اُس ہستی15 کی معرفت عدل سے کرےگا جسے اُس نے اِسی کام کے لئے مقرر کیا ہے۔ اِس کا ثبوت خدا نے تمام لوگوں کو یہ دیا ہے اسرائیل کے بہت سے لوگوں کے لئے ہدایت اور بعض اشخاص کے لئے بربادی کا سبب بنےگا۔ وہ خدا کی ایسی نشانی بنےگا جس کے بہت سے لوگ مخالف ہو جائیں‌گے۔ یوں اُن کے دلوں کے مذموم راز کھل جائیں‌گے“ (آیۃُ اللّٰه 2‏:34)۔ موجودہ صورتِ حال میں کچھ لوگ تو منکرین میں شامل ہو گئے اور مولا عیسٰیؑ کی شان میں گستاخی کرنے لگے اور کچھ آپؑ پر ایمان لے آئے اور اُنھیں قربِ الٰہی کی سعادت نصیب ہوئی۔ نیز دیکھئے: ”…پس بنی عربی مصادر میں اِس کے پیروکار ”رَوَّاقِیِّین“ اور فلسفہ ”رَوَّاقِیَّة“ کہلاتا ہے۔11سننے والوں کے ذہن میں یہ غلط‌فہمی اِس لئے پیدا ہوئی کیوں کہ یونانی میں ”عیسٰی“ (ἰησοῦς) مذکر ہے اور ”قیامت“ یا ”دوبارہ زندہ ہو جانا“ (ἀνάστασις) مؤنث۔ اپنے مشرکانہ پس منظر کی روشنی میں وہ لوگ یہ سمجھ بیٹھے کہ حضرت طالوتِ اصغرؓ — نعوذ باللّٰه! — کسی ”عیسٰی“ دیوتا اور ”قیامت“ دیوی کی بات کر رہے ہیں۔12نیز دیکھئے: ”اور ہر گروہ کے لئے ایک میعاد مقرر ہے…“ (قرآنِ مجید، سورۃ الاعراف [7]، آیت 34)۔13چھٹی صدی قبلِ مسیح کے ایک شاعر اَیپی‌مَیْنِدیس (Epimenides) سے اقتباس۔ اِسی شاعر نے یہ مشورہ دیا تھا کہ ایک استھان ”نامعلوم دیوتا“ (دیکھئے آیت 23) کے نام پر بھی ہونا چاہئے تاکہ کوئی دیوتا ناراض نہ رہے۔14اَراطُس (Aratus) کے کلام سے اقتباس۔ ویسے یہی بات ایک روّاقی فلسفی کْلیئَنتھیس (Cleanthes) نے بھی کہی ہے۔15یعنی خلیفۃُ اللّٰه سیدنا عیسٰیؑ جو اِس دنیا میں عدل کرنے اور عدل قائم کرنے آئیں‌گے۔ جس بات کی پیش‌گوئی حضرت صفوانؓ نے کی تھی (10‏:42)، اُس کا تذکرہ حضرت طالوتِ اصغرؓ یہاں فرماتے ہیں۔ دونوں صحابۂ کرامؓ کی تعلیمات میں کوئی فرق یا تضاد نہیں۔16مرنے کے بعد قیامت کے دن زندہ کیا جانا تو بنیادی عقیدے کا ایک جزو ہے، مگر بت‌پرست کافروں نے اِس کا بھی انکار کیا۔ اُن کے برعکس، ہمارا عقیدہ ”البعث بعد الموت“ (یعنی مرنے کے بعد جی اٹھنے) پر ہے۔17سخت سے سخت کافروں کے درمیان بھی نورِ ہدایت چمکتا اور اپنا کام کرتا ہے۔ خدا کے نشانات و آیات اُن لوگوں کے لئے کافی ہیں جو ایمان لانا چاہتے ہیں۔18اصل زبان میں διονύσιος [”دِیونُسیوس“]۔ خداوندِ خمر διόνυσος [”دِیونُسوس“] کے نام پر اِن کا نام رکھا گیا تھا۔ اِن صاحب کا ذکر صرف یہیں آیا ہے اور اِن کے بارے میں مزید کچھ معلوم نہیں۔19اللّٰه تعالٰی کسی کا اجر
